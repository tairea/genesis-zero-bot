# graph.regentribe.org — Live Knowledge Graph
# Serves the 3D visualization directly and proxies WebSocket to live_server.py

server {
    listen 80;
    server_name graph.regentribe.org;
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name graph.regentribe.org;

    ssl_certificate /etc/letsencrypt/live/graph.regentribe.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/graph.regentribe.org/privkey.pem;

    # WebSocket proxy to live_server.py
    location /ws {
        proxy_pass http://127.0.0.1:8788;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_connect_timeout 300s;
        proxy_send_timeout 300s;
        proxy_read_timeout 300s;
        proxy_buffering off;
    }

    # Health check
    location /health {
        return 200 'ok';
        add_header Content-Type text/plain;
    }

    # Serve the live visualization at /
    root /home/ian/.openclaw/workspace-genesis/skills/semantic-graph/viz;
    index live.html;

    location / {
        try_files $uri /live.html;
    }
}
